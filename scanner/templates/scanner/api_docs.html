Usage Examples
Create Project & Scan
# Create project
curl -X POST http://127.0.0.1:8000/api/projects/ \
  -H "Content-Type: application/json" \
  -d '{"name":"Target","domain":"example.com"}'

# Start async scan
curl -X POST http://127.0.0.1:8000/api/projects/1/scan/
Query Results
# Get subdomains for scan 1
curl http://127.0.0.1:8000/api/scans/1/subdomains/

# Filter critical nuclei findings
curl "http://127.0.0.1:8000/api/scans/1/nuclei/?severity=critical"

# Diff two scans
curl "http://127.0.0.1:8000/api/projects/1/diff/?scan_a=1&scan_b=2"
